Sample SQL Queries

Basic SELECT

select * from users where age > 18 and status = 'active' order by created_at desc;

Complex JOIN

select u.name, p.title, c.name as category from users u inner join posts p on u.id = p.user_id left join categories c on p.category_id = c.id where u.active = 1;

Subquery

select * from products where price > (select avg(price) from products where category_id = 1) and stock_quantity > 0;

INSERT Statement

insert into users (name, email, age, created_at) values ('John Doe', 'john@example.com', 25, now()), ('Jane Smith', 'jane@example.com', 30, now());

UPDATE Statement

update users set last_login = now(), login_count = login_count + 1 where email = 'user@example.com' and active = 1;

CREATE TABLE

create table orders (id int primary key auto_increment, user_id int not null, total decimal(10,2), status varchar(20) default 'pending', created_at timestamp default current_timestamp);
